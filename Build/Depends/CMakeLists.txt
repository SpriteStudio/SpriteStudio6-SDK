macro(Depends_Lib DEPSDIR)
    message(${DEPSDIR})

    # glew include directories
    if (DEFINED $ENV{GLEW_INCLUDE_PATH})
        include_directories($ENV{GLEW_INCLUDE_PATH})
    else()
        if(MSVC)
            include_directories(${DEPSDIR}/glew-2.1.0/include)
        endif()
    endif()
    ## glew library directories
    if (DEFINED $ENV{GLEW_LIB_PATH})
        set(GLEW_LIB_PATH $ENV{GLEW_LIB_PATH})
    else()
        if(WIN32)
            set(GLEW_LIB_PATH ${DEPSDIR}/glew-2.1.0/lib/Release/Win32)
        elseif(WIN64)
            set(GLEW_LIB_PATH ${DEPSDIR}/glew-2.1.0/lib/Release/x64)
        endif()
    endif()
    link_directories(${GLEW_LIB_PATH})

    # glfw3 include directories
    if (DEFINED $ENV{GLFW_INCLUDE_PATH})
        include_directories($ENV{GLFW_INCLUDE_PATH})
    else()
        if(WIN32)
            set(GLFW_LIB_ARCH 32)
            include_directories(${DEPSDIR}/glfw-3.2.1.bin.WIN${GLFW_LIB_ARCH}/include)
        elseif(WIN64)
            set(GLFW_LIB_ARCH 64)
            include_directories(${DEPSDIR}/glfw-3.2.1.bin.WIN${GLFW_LIB_ARCH}/include)
        elseif(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            include_directories(${DEPSDIR}/glfw/include)
        endif()
    endif()
    ## glfw3 library directories
    if (DEFINED $ENV{GLFW_LIB_PATH})
        set(GLFW_LIB_PATH $ENV{GLFW_LIB_PATH})
    else()
        # TODO: change msvc version
        if(MSVC)
            set(GLFW_LIB_PATH ${DEPSDIR}/glfw-3.2.1.bin.WIN${GLFW_LIB_ARCH}/lib-vc2015)
        elseif(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            set(GLFW_LIB_PATH ${DEPSDIR}/glfw/lib)
        endif()
    endif()
    link_directories(${GLFW_LIB_PATH})

    # anttweakbar include directories
    if (DEFINED $ENV{ANTTWEAKBAR_INCLUDE_PATH})
        include_directories($ENV{ANTTWEAKBAR_INCLUDE_PATH})
    else()
        if(MSVC)
            include_directories(${DEPSDIR}/AntTweakBar/include)
        elseif(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            include_directories(${DEPSDIR}/anttweakbar/include)
        endif()
    endif()
    # anttweakbar library directories
    if (DEFINED $ENV{ANTTWEAKBAR_LIB_PATH})
        set(ANTTWEAKBAR_LIB_PATH $ENV{ANTTWEAKBAR_LIB_PATH})
    else()
        if(WIN32)
            set(ANTTWEAKBAR_LIB_PATH ${DEPSDIR}/AntTweakBar/lib)
            set(ANTTWEAKBAR_LIB_ARCH)
        elseif(WIN64)
            # don't consider ARM yet.
            set(ANTTWEAKBAR_LIB_PATH ${DEPSDIR}/AntTweakBar/lib)
            set(ANTTWEAKBAR_LIB_ARCH 64)
        elseif(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            set(ANTTWEAKBAR_LIB_PATH ${DEPSDIR}/anttweakbar/lib)
        endif()
    endif()
    link_directories(${ANTTWEAKBAR_LIB_PATH})

    # boost include directories
    if (DEFINED $ENV{BOOST_INCLUDE_PATH})
        include_directories($ENV{BOOST_INCLUDE_PATH})
    else()
        if(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            include_directories(${DEPSDIR}/boost/include)
        elseif(MSVC)
            include_directories(${DEPSDIR}/boost_1_66_0)
        endif()
    endif()

    ## boost library directories
    if (DEFINED $ENV{BOOST_LIB_PATH})
        set(BOOST_LIB_PATH $ENV{BOOST_LIB_PATH})
    else()
        if(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            set(BOOST_LIB_PATH ${DEPSDIR}/boost/lib)
        else(MSVC)
            # TODO: change msvc version
            if(WIN32)
                set(BOOST_LIB_PATH ${DEPSDIR}/boost_1_66_0/lib32-msvc-14.1)
                set(BOOST_LIB_ARCH x32)
            else()
                # don't consider ARM yet.
                set(BOOST_LIB_PATH ${DEPSDIR}/boost_1_66_0/lib64-msvc-14.1)
                set(BOOST_LIB_ARCH x64)
            endif()
        endif()
    endif()
    link_directories(${BOOST_LIB_PATH})
    if(MSVC AND ENABLE_STATIC)
        # TODO: change msvc version
        if(CMAKE_BUILD_TYPE MATCHES Release)
            set(BOOST_LIB_SUFFIX vc141-mt-${BOOST_LIB_ARCH}-1_66)
        else()
            set(BOOST_LIB_SUFFIX vc141-mt-gd-${BOOST_LIB_ARCH}-1_66)
        endif()
    else()
    # TODO:
    endif()

    # python include directories
    if (DEFINED $ENV{PYTHON_INCLUDE_PATH})
        include_directories($ENV{PYTHON_INCLUDE_PATH})
    else()
        if(APPLE AND (${CMAKE_SYSTEM_NAME} MATCHES "Darwin"))
            # TODO:
            include_directories(${DEPSDIR}/python/Frameworks/Python.framework/Versions/3.7/Headers)
        elseif(MSVC)
            include_directories(${DEPSDIR}/boost_1_66_0)
        endif()
    endif()
endmacro()
