55class VarianceValue : public SsXmlRangeValueConverter
56{
57private:
58
59 enum RangeType { None, MinMax, PlusMinus,};
60
61 RangeType type;
62 mytype value;
63 mytype subvalue;
64
65
66public:
67
68 VarianceValue( mytype v ){
69 value = v;
70 subvalue = v;
71 type = VarianceValue::None;
72 }
73 VarianceValue( mytype v , mytype v2){
74 value = v;
75 subvalue = v2;
76 type = VarianceValue::MinMax;
77 }
78
79 void setPlusMinus( mytype v1 )
80 {
81 value = -v1;
82 subvalue = v1;
83 type = VarianceValue::PlusMinus;
84 }
85
86 void setMinMax(mytype min , mytype max)
87 {
88 value = min;
89 subvalue = max;
90 type = VarianceValue::MinMax;
91 }
92
93 mytype getValue(){ return value;}
94 mytype getMinValue(){ return value;}
95 mytype getMaxValue(){ return subvalue;}
96
97 bool isTypeNone(){ return type == VarianceValue::None; }
98 bool isTypeMinMax(){ return type == VarianceValue::MinMax; }
99 bool isTypePlusMinus(){ return type == VarianceValue::PlusMinus; }
100
101 mytype* getlpValue(){ return &value;}
102 mytype* getlpSubValue(){ return &subvalue;}
103
104 VarianceValue& operator=(mytype v) { value = v; return *this; }
105 operator mytype() { return value; }
106
107private:
108 virtual bool inputString( SsString value , SsString subvalue )
109 {
110 return true;
111 }
112};
